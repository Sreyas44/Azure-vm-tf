    name = "sreyas-nic"

        name = "sreyas-subnet"

    name                       = "Allow_HTTP"

    source_address_prefix      = "*"
    destination_address_prefix = "*"

  security_rule {
    name                       = "Allow_RDP"
    priority                   = 100
    direction                  = "Inbound"
